private void btnUpdateAdmin_Click(object sender, EventArgs e)
{
    try
    {
        if (!this.IsValidToAdd())
        {
            MessageBox.Show("Please select a product and fill all fields");
            return;
        }
        var query = "select * from tblLogin where Id ='" + this.txtUserId.Text + "';";
        var dt = this.dbc.ExecuteQueryTable(query);

        if (dt.Rows.Count == 0)
        {
            MessageBox.Show("User does not exist.");
            return;
        }
        //update
        var sql = @"update tblLogin
                        set Username = '" + this.txtUserName.Text + @"',
                        Password = '" + txtPassword.Text + @"'
                        where Id = '" + this.txtUserId.Text + "';";
        var count = this.dbc.ExecuteDMLQuery(sql);

        if (count == 1)
            MessageBox.Show("Admin has been updated");
        else
            MessageBox.Show("Admin hasn't been updated");

        this.PopulateGridView();
        this.ClearAll();
    }
    catch (Exception exc)
    {
        MessageBox.Show("An error has occured: " + exc.Message);
    }
}
